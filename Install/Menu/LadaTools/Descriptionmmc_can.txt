int Init(HWND m_MsgHWND); // Функция инициализации. Открывает порт и включает таймер. Возвращает 1 при удачном выполнении. m_MsgHWND - HWND которому будут отправляться сообщения. Если m_MsgHWND = 0, то сообщения будут отсылаться всем.
float GetEngineTemp(); // Функция возвращает значение температуры двигателя
float GetFuel(); // Функция возвращает значение уровня топлива
float GetGearboxTemp(); // Функция возвращает значение температуры АКПП
float GetAirTemp(); // Функция возвращает значение температуры воздуха (при наличии датчика внешней температуры)
int GetBatteryChargeFlag(); // Функция возвращает значение флага заряда АБК
int GetWarningEngineTempFlag(); // Функция возвращает значение флага превышения температуры двигателя
int GetWarningOilPressureFlag(); // Функция возвращает значение флага аварийного давления масла
int GetCheckEngine(); // Функция возвращает значение флагов Check Engine (2 бита)
int GetActiveCAN(); // Функция возвращает значение 1 если сообщения от комбинации приборов принимаются (вкл. зажигание), 0 - если не принимаются.
int ResetECUErrors(); // Функция посылает запрос на сброс ошибок в КСУД
int ResetEngineECU(int mode); // Функция выполняет запрос на сброс КСУД: mode = 1  - простой сброс как при выключении питания
                                                                         mode = 144  - полный сброс к заводским установкам
                                                                         mode = 145  - сброс обучения
